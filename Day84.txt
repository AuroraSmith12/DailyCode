//SCL PB8
//SDA PB9

void OLED_YiZhi(void)
{
	 RCC_APB2PeriphClockCmd( RCC_APB2Periph_GPIOB,  ENABLE);

	
	GPIO_InitTypeDef SCL;
	SCL.GPIO_Mode=GPIO_Mode_Out_PP;
	SCL.GPIO_Pin=GPIO_Pin_8;
	SCL.GPIO_Speed=GPIO_Speed_50MHz;
	
	GPIO_InitTypeDef SDA;
	SDA.GPIO_Mode=GPIO_Pin_9;
	SDA.GPIO_Pin=GPIO_Mode_Out_PP;
	SDA.GPIO_Speed=GPIO_Speed_50MHz;
	
	 GPIO_Init(GPIOB,  &SCL);
	 GPIO_Init(GPIOB,  &SDA);

	
}

void IIC_Start()
{
	 GPIO_ResetBits( GPIOB,  GPIO_Pin_8);

	GPIO_SetBits( GPIOB,  GPIO_Pin_9);
 GPIO_SetBits( GPIOB,  GPIO_Pin_8);
	Delay_us(5);
	GPIO_ResetBits( GPIOB,  GPIO_Pin_9);
	Delay_us(5);
}

void IIC_Stop()
{
	 GPIO_ResetBits( GPIOB,  GPIO_Pin_8);
	GPIO_ResetBits( GPIOB,  GPIO_Pin_9);
 GPIO_SetBits( GPIOB,  GPIO_Pin_8);
	Delay_us(5);
	GPIO_SetBits( GPIOB,  GPIO_Pin_9);
	Delay_us(5);
}

char IIC_ACK()
{
	char flag;
	GPIO_SetBits( GPIOB,  GPIO_Pin_9);//就在时钟脉冲9期间释放数据线
	Delay_us(5);
 GPIO_SetBits( GPIOB,  GPIO_Pin_8);
  Delay_us(5);
	flag =  GPIO_ReadOutputDataBit( GPIOB,  GPIO_Pin_8);
	Delay_us(5);
	 GPIO_ResetBits( GPIOB,  GPIO_Pin_8);
	Delay_us(5);
	
	return flag;
}

void IIC_Send_Byte(char dataSend)
{
	int i;
	
	for(i = 0;i<8;i++){
		 GPIO_ResetBits( GPIOB,  GPIO_Pin_8);//scl拉低，让sda做好数据准备
	
 

	 GPIO_WriteBit(GPIOB,  GPIO_Pin_9,  dataSend & 0x80);//1000 0000获得dataSend的最高位，给sda
		Delay_us(5);//发送数据建立时间
		 GPIO_SetBits( GPIOB,  GPIO_Pin_8);//scl拉高开始发送
		Delay_us(5);//数据发送时间
			 GPIO_ResetBits( GPIOB,  GPIO_Pin_8);//发送完毕拉低
		Delay_us(5);//
		dataSend = dataSend << 1;
	}
}

void Oled_Write_Cmd(char dataCmd)
{
	//	1. start()
	IIC_Start();
	//		
	//	2. 写入从机地址  b0111 1000 0x78
	IIC_Send_Byte(0x78);
	//	3. ACK
	IIC_ACK();
	//	4. cotrol byte: (0)(0)000000 写入命令   (0)(1)000000写入数据
	IIC_Send_Byte(0x00);
	//	5. ACK
	IIC_ACK();
	//6. 写入指令/数据
	IIC_Send_Byte(dataCmd);
	//7. ACK
	IIC_ACK();
	//8. STOP
	IIC_Stop();
}

void Oled_Write_Data(char dataData)
{
	//	1. start()
	IIC_Start();
	//		
	//	2. 写入从机地址  b0111 1000 0x78
	IIC_Send_Byte(0x78);
	//	3. ACK
	IIC_ACK();
	//	4. cotrol byte: (0)(0)000000 写入命令   (0)(1)000000写入数据
	IIC_Send_Byte(0x40);
	//	5. ACK
	IIC_ACK();
	///6. 写入指令/数据
	IIC_Send_Byte(dataData);
	//7. ACK
	IIC_ACK();
	//8. STOP
	IIC_Stop();
}


void Oled_Init(void){
	Oled_Write_Cmd(0xAE);//--display off
	Oled_Write_Cmd(0x00);//---set low column address
	Oled_Write_Cmd(0x10);//---set high column address
	Oled_Write_Cmd(0x40);//--set start line address  
	Oled_Write_Cmd(0xB0);//--set page address
	Oled_Write_Cmd(0x81); // contract control
	Oled_Write_Cmd(0xFF);//--128   
	Oled_Write_Cmd(0xA1);//set segment remap 
	Oled_Write_Cmd(0xA6);//--normal / reverse
	Oled_Write_Cmd(0xA8);//--set multiplex ratio(1 to 64)
	Oled_Write_Cmd(0x3F);//--1/32 duty
	Oled_Write_Cmd(0xC8);//Com scan direction
	Oled_Write_Cmd(0xD3);//-set display offset
	Oled_Write_Cmd(0x00);//
	
	Oled_Write_Cmd(0xD5);//set osc division
	Oled_Write_Cmd(0x80);//
	
	Oled_Write_Cmd(0xD8);//set area color mode off
	Oled_Write_Cmd(0x05);//
	
	Oled_Write_Cmd(0xD9);//Set Pre-Charge Period
	Oled_Write_Cmd(0xF1);//
	
	Oled_Write_Cmd(0xDA);//set com pin configuartion
	Oled_Write_Cmd(0x12);//
	
	Oled_Write_Cmd(0xDB);//set Vcomh
	Oled_Write_Cmd(0x30);//
	
	Oled_Write_Cmd(0x8D);//set charge pump enable
	Oled_Write_Cmd(0x14);//
	
	Oled_Write_Cmd(0xAF);//--turn on oled panel		
}

void Oled_Clear()
{
	unsigned char i,j; //-128 --- 127
	
	for(i=0;i<8;i++){
		Oled_Write_Cmd(0xB0 + i);//page0--page7
		//每个page从0列
		Oled_Write_Cmd(0x00);
		Oled_Write_Cmd(0x10);
		//0到127列，依次写入0，每写入数据，列地址自动偏移
		for(j = 0;j<128;j++){
			Oled_Write_Data(0);
		}
	}
}

void Oled_Show_Image(unsigned char *image)
{
	unsigned char i; 
	unsigned int j;
	
	for(i=0;i<8;i++){
		Oled_Write_Cmd(0xB0 + i);//page0--page7
		//每个page从0列
		Oled_Write_Cmd(0x00);
		Oled_Write_Cmd(0x10);
		//0到127列，依次写入0，每写入数据，列地址自动偏移
		for(j = 128 * i; j<(128 * (i+1));j++){
			Oled_Write_Data(image[j]);
		}
	}
}
